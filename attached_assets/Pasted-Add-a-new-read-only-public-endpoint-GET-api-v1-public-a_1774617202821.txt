def list_all_assets(
    asset_class: Optional[str] = Query(None, description="Filter by asset class: forex, crypto, stocks"),
    strategy: Optional[str] = Query(None, description="Filter by strategy name e.g. mtf_ema"),
    category: Optional[str] = Query(None, description="Filter by category label: Forex, Crypto, Commodities, Indices, Stocks, ETFs"),
):
    """
    Returns the full catalogue of all active tradeable assets
    across all strategies. No signal data — purely asset metadata:
    symbol, full name, asset class, category label, and strategies.
    """
    from trading_engine.database import get_strategy_assets_full, get_full_name_for_asset

    # User-friendly category mapping based on symbol and sub_category
    def _resolve_category(symbol: str, asset_class: str, sub_category: str) -> str:
        if asset_class == "stocks" or sub_category == "nasdaq100":
            return "Stocks"
        if asset_class == "crypto" or sub_category == "crypto":
            return "Crypto"
        if sub_category == "indices":
            return "Indices"
        if sub_category == "commodities":
            return "Commodities"
        # ETF symbols — no slash, not an index, not crypto
        if "/" not in symbol and asset_class == "forex":
            return "ETFs"
        # Forex pairs — contain a slash
        if "/" in symbol and asset_class == "forex":
            return "Forex"
        return "Other"

    rows = get_strategy_assets_full()

    # Deduplicate by symbol — merge strategy names into a list
    assets: dict[str, dict] = {}
    for row in rows:
        if not (row.get("is_active") is True or row.get("is_active") == 1):
            continue
        symbol = row["symbol"]
        asset_class_val = row.get("asset_class", "other")
        sub_category_val = row.get("sub_category") or ""
        category_label = _resolve_category(symbol, asset_class_val, sub_category_val)

        if symbol not in assets:
            assets[symbol] = {
                "symbol": symbol,
                "full_name": row.get("full_name") or get_full_name_for_asset(symbol),
                "asset_class": asset_class_val,
                "sub_category": sub_category_val or None,
                "category": category_label,
                "strategies": [],
            }
        strat = row.get("strategy_name")
        if strat and strat not in assets[symbol]["strategies"]:
            assets[symbol]["strategies"].append(strat)

    # Apply filters
    result = list(assets.values())

    if asset_class:
        result = [a for a in result if a["asset_class"] == asset_class.lower()]

    if strategy:
        result = [a for a in result if strategy in a["strategies"]]

    if category:
        result = [a for a in result if a["category"].lower() == category.lower()]

    # Sort: by category then symbol
    CATEGORY_ORDER = {
        "Forex": 1,
        "Crypto": 2,
        "Commodities": 3,
        "Indices": 4,
        "ETFs": 5,
        "Stocks": 6,
        "Other": 7,
    }
    result = sorted(
        result,
        key=lambda x: (CATEGORY_ORDER.get(x["category"], 99), x["symbol"])
    )

    return {
        "count": len(result),
        "categories": sorted(list({a["category"] for a in result}), key=lambda c: CATEGORY_ORDER.get(c, 99)),
        "assets": result,
    }
