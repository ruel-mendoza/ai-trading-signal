PART 2 — NEW ADMIN API ENDPOINT (admin.py)
─────────────────────────────────────────────────────────────────

Add a new endpoint to admin.py:

POST /admin/api/strategy-assets/verify

Body:
{
  "strategy_name": "stocks_algo1",
  "symbol": "AAPL"
}

Logic:

1. Validate strategy_name and symbol are present
   If missing: return 400 {"success": False, 
   "error": "strategy_name and symbol are required"}

2. Fetch the asset row from DB:
   rows = get_strategy_assets_full(strategy_name)
   Find the row matching symbol.
   If not found: return 404 {"success": False,
   "error": f"{symbol} not found in {strategy_name}"}

3. Run verification based on strategy type:

   For stocks_algo1 and stocks_algo2:
   Use the dedicated _verify_stock_symbol() function
   (defined below — add as a module-level helper in admin.py)

   For all other strategies:
   Use engine.cache.api_client.test_symbol_coverage(symbol)
   Get engine via:
     from trading_engine import engine_registry
     engine = engine_registry.get_engine()
     if engine is None:
         return 503 {"success": False, 
         "error": "Engine not ready — retry in a few seconds"}

4. If verification passes (supported=True):
   - Call mark_asset_verified(strategy_name, symbol, verified=True)
   - Return 200:
     {
       "success": True,
       "symbol": symbol,
       "strategy_name": strategy_name,
       "verified": True,
       "sample_close": result.get("sample_close"),
       "sample_time": result.get("sample_time"),
       "candles_returned": result.get("candles_returned"),
       "asset_class": result.get("asset_class"),
       "message": f"{symbol} verified successfully against FCSAPI"
     }

5. If verification fails (supported=False):
   - Call mark_asset_verified(strategy_name, symbol, verified=False)
   - Return 200 (not 4xx — the request succeeded, the asset
     just failed verification):
     {
       "success": False,
       "symbol": symbol,
       "strategy_name": strategy_name,
       "verified": False,
       "reason": result.get("reason", "Not supported"),
       "message": f"{symbol} failed FCSAPI verification: 
                   {result.get('reason')}"
     }

6. Log every verification attempt at INFO level:
   [ADMIN] Verify asset: {strategy_name}/{symbol} | 
   result={supported} | by={username}

def _verify_stock_symbol(symbol: str, api_client) -> dict:
    """
    Verify a NASDAQ equity symbol using the exact same endpoint
    and parameters as _fetch_stock_candles() in stocks_algo1.py
    and stocks_algo2.py. Ensures verification accurately reflects
    whether the strategy can actually fetch data for this symbol.
    """
    from trading_engine.fcsapi_client import (
        BASE_URL_V4_STOCK,
        TIMEFRAME_MAP,
        _parse_response_items,
        _validate_candle_prices,
    )
    from trading_engine.credit_control import pre_request_check
    from trading_engine.database import log_api_usage
    import requests as _req

    api_key = api_client.api_key
    if not api_key:
        return {
            "supported": False,
            "reason": "No API key configured"
        }

    params = {
        "symbol": symbol,
        "period": TIMEFRAME_MAP.get("D1", "1d"),
        "length": "2",
        "type": "equity",
        "exchange": "NASDAQ",
        "access_key": api_key,
    }

    try:
        pre_request_check()
        url = f"{BASE_URL_V4_STOCK}/history"
        resp = _req.get(url, params=params, timeout=30)
        resp.raise_for_status()
        data = resp.json()
        log_api_usage(endpoint=f"stock/history/verify/{symbol}")

        if not data.get("status") or not data.get("response"):
            return {
                "supported": False,
                "symbol": symbol,
                "reason": data.get("msg", "No data returned"),
            }

        candles = _parse_response_items(data["response"])
        candles = _validate_candle_prices(candles, symbol)

        if not candles:
            return {
                "supported": False,
                "symbol": symbol,
                "reason": "API returned data but no valid candles",
            }

        return {
            "supported": True,
            "symbol": symbol,
            "api_symbol": symbol,
            "asset_class": "stocks",
            "base_url": BASE_URL_V4_STOCK,
            "sample_close": candles[-1]["close"],
            "sample_time": candles[-1]["timestamp"],
            "candles_returned": len(candles),
        }

    except Exception as e:
        return {
            "supported": False,
            "symbol": symbol,
            "reason": str(e),
        }

showVerifyToast(message, type, duration)
────────────────────────────────────

Creates and auto-dismisses a toast notification:

  type: 'success' | 'error' | 'info'
  duration: milliseconds before auto-dismiss (default 3000)

Implementation:

function showVerifyToast(message, type, duration) {
    duration = duration || 3000;

    // Remove any existing toast
    var existing = document.getElementById('verify-toast-el');
    if (existing) existing.remove();

    var icons = {
        success: '✓',
        error:   '✗',
        info:    'ⓘ'
    };

    var toast = document.createElement('div');
    toast.id = 'verify-toast-el';
    toast.className = 'verify-toast ' + type;
    toast.setAttribute('data-testid', 'toast-verify-result');
    toast.innerHTML =
        '<span style="font-size:16px;">' +
        (icons[type] || 'ⓘ') + '</span>' +
        '<span>' + message + '</span>';

    document.body.appendChild(toast);

    // Auto-dismiss
    setTimeout(function() {
        if (toast.parentNode) {
            toast.style.animation =
                'fadeOut 0.4s ease forwards';
            setTimeout(function() {
                if (toast.parentNode) toast.remove();
            }, 400);
        }
    }, duration);
}

async function verifyAsset(strategyName, symbol, btnEl) {
    // Update button to loading state
    btnEl.disabled = true;
    btnEl.textContent = '⏳ Verifying...';
    btnEl.className = 'btn-verify verifying';

    // Show info toast while verifying
    showVerifyToast(
        'Verifying ' + symbol + ' against FCSAPI...',
        'info',
        10000
    );

    try {
        var res = await fetch(
            BASE + '/admin/api/strategy-assets/verify',
            {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    strategy_name: strategyName,
                    symbol: symbol
                })
            }
        );
        var data = await res.json();

        if (data.success && data.verified) {
            // Update button to verified state
            btnEl.textContent = '✓ Verified';
            btnEl.className = 'btn-verify verified';
            btnEl.disabled = true;
            btnEl.style.cursor = 'default';

            // Show success toast with sample data
            var detail = '';
            if (data.sample_close) {
                detail = ' | Close: ' + data.sample_close;
            }
            if (data.candles_returned) {
                detail += ' | ' + data.candles_returned + 
                          ' candles';
            }
            showVerifyToast(
                '✓ ' + symbol + ' verified successfully' + 
                detail,
                'success',
                4000
            );

            // Update the badge in the table cell
            // without reloading the whole table
            var cell = btnEl.closest('td');
            if (cell) {
                cell.innerHTML =
                    '<span class="badge status-active" ' +
                    'data-testid="badge-verified-' +
                    symbol.replace(/\//g, '-') + '">' +
                    '✓ Verified</span>';
            }

        } else {
            // Update button to failed state
            btnEl.disabled = false;
            btnEl.textContent = '✗ Failed — Retry';
            btnEl.className = 'btn-verify failed';

            showVerifyToast(
                '✗ ' + symbol + ' verification failed: ' +
                (data.reason || data.message || 
                 'Not supported on current plan'),
                'error',
                6000
            );
        }

    } catch (err) {
        btnEl.disabled = false;
        btnEl.textContent = '⚡ Verify';
        btnEl.className = 'btn-verify';

        showVerifyToast(
            '✗ Network error verifying ' + symbol + 
            ': ' + err.message,
            'error',
            5000
        );
    }
}

async function verifyAllUnverified() {
    var buttons = document.querySelectorAll(
        '.btn-verify:not(:disabled)'
    );
    if (buttons.length === 0) {
        showVerifyToast(
            'All assets are already verified', 
            'info', 
            3000
        );
        return;
    }

    var confirmed = confirm(
        'Verify ' + buttons.length + ' unverified asset(s)?' +
        '\n\nThis will use ' + buttons.length + 
        ' FCSAPI credit(s).'
    );
    if (!confirmed) return;

    showVerifyToast(
        'Verifying ' + buttons.length + 
        ' asset(s) — please wait...',
        'info',
        buttons.length * 2000
    );

    var success = 0;
    var failed  = 0;

    for (var i = 0; i < buttons.length; i++) {
        var btn = buttons[i];
        var strategy = btn.dataset.strategy;
        var symbol   = btn.dataset.symbol;

        await verifyAsset(strategy, symbol, btn);

        // Read result from button class
        if (btn.classList.contains('verified')) {
            success++;
        } else {
            failed++;
        }

        // Delay between calls to avoid rate limiting
        if (i < buttons.length - 1) {
            await new Promise(r => setTimeout(r, 1500));
        }
    }

    showVerifyToast(
        'Batch verify complete: ' + success + 
        ' passed, ' + failed + ' failed',
        failed > 0 ? 'error' : 'success',
        5000
    );
}
