def get_nasdaq_api_symbol(symbol: str) -> str:
    """
    Convert a clean NASDAQ ticker to the FCSAPI v4 format.

    FCSAPI v4 requires exchange-prefixed symbols for NASDAQ
    equity stocks:
      AAPL  →  NASDAQ:AAPL
      MSFT  →  NASDAQ:MSFT
      NVDA  →  NASDAQ:NVDA

    Symbols that already contain a colon or slash are returned
    unchanged — this covers forex (EUR/USD), crypto (BTC/USD),
    and any symbol already prefixed (NASDAQ:AAPL).

    This function must ONLY be used for plain NASDAQ equity
    tickers. Do NOT apply it to:
      - ETF symbols (CORN, USO, SGOL) — handled by ETF routing
      - Index symbols (SPX, NDX, DJI) — in STOCK_INDEX_SYMBOLS
      - Crypto symbols (BTC/USD) — contain slash
      - Forex symbols (EUR/USD) — contain slash
      - Commodity symbols (XAU/USD) — contain slash
    """
    if ":" in symbol or "/" in symbol:
        return symbol
    return f"NASDAQ:{symbol}"

def _fetch_stock_candles(
    cache: CacheLayer, symbol: str, limit: int = 300
) -> list[dict]:
    """
    Fetch D1 candles for a NASDAQ-listed equity via FCSAPI v4.

    Uses NASDAQ: prefix format as required by FCSAPI v4:
      AAPL → NASDAQ:AAPL
      MSFT → NASDAQ:MSFT

    The prefix is applied by get_nasdaq_api_symbol() and is
    only added to plain ticker symbols — symbols containing
    slashes or colons are passed through unchanged.
    """
    from trading_engine.fcsapi_client import (
        BASE_URL_V4_STOCK,
        TIMEFRAME_MAP,
        get_nasdaq_api_symbol,
        _parse_response_items,
        _validate_candle_prices,
    )
    from trading_engine.credit_control import pre_request_check
    from trading_engine.database import log_api_usage
    import requests as _req

    api_key = cache.api_client.api_key
    if not api_key:
        logger.warning(
            f"[ALGO1] No API key — "
            f"cannot fetch candles for {symbol}"
        )
        return []

    # Apply NASDAQ: prefix — key fix for FCSAPI v4
    api_symbol = get_nasdaq_api_symbol(symbol)
    tf_api     = TIMEFRAME_MAP.get("D1", "1d")
    url        = f"{BASE_URL_V4_STOCK}/history"

    params = {
        "symbol":     api_symbol,   # NASDAQ:AAPL not AAPL
        "period":     tf_api,
        "length":     str(limit),
        "access_key": api_key,
    }

    logger.info(
        f"[ALGO1] Fetching candles | "
        f"symbol={symbol} | "
        f"api_symbol={api_symbol} | "
        f"period={tf_api} | "
        f"length={limit}"
    )

    try:
        pre_request_check()
        resp = _req.get(url, params=params, timeout=30)

        if resp.status_code == 400:
            logger.error(
                f"[ALGO1] {symbol} | "
                f"400 Bad Request | "
                f"api_symbol={api_symbol} | "
                f"body={resp.text[:200]}"
            )
            return []

        resp.raise_for_status()
        data = resp.json()
        log_api_usage(
            endpoint=f"stock/history/{symbol}"
        )

        if not data.get("status") or not data.get("response"):
            logger.warning(
                f"[ALGO1] {symbol} | "
                f"No data returned | "
                f"msg={data.get('msg', '')}"
            )
            return []

        candles = _parse_response_items(data["response"])
        candles = _validate_candle_prices(candles, symbol)

        logger.info(
            f"[ALGO1] {symbol} | "
            f"✓ {len(candles)} candles returned | "
            f"latest_close="
            f"{candles[-1]['close'] if candles else 'N/A'}"
        )
        return candles

    except Exception as e:
        logger.warning(
            f"[ALGO1] {symbol} | "
            f"Candle fetch failed: {e}"
        )
        return []
