=============================================================

Exit logic only runs inside evaluate() when open_position_data is

calls evaluate() per asset but NEVER calls check_exits() after the loop.

def has_any_open_signal_for_asset(asset: str) -> bool:
    """
    Returns True if ANY open signal exists for this asset across
    ALL strategies. Used as the primary cross-strategy idempotency
    guard to prevent simultaneous BUY+SELL on the same asset.
    """
    with _get_session() as session:

                Signal.direction,
                Signal.entry_price,
                Signal.created_at,
            )

                Signal.asset   == asset,
                Signal.status  == "OPEN",
            )
            .order_by(Signal.id.desc())
            .first()
        )

                f"BLOCKED by signal #{row[0]} | "
                f"strategy={row[1]} | direction={row[2]} | "
                f"entry={row[3]} | created={row[4]}"
            )
            return True
        return False

def cleanup_orphan_records() -> dict:
    """
    Find and fix orphan signals (OPEN signal with no position record)
    and orphan positions (position with no OPEN signal).
    
    Orphan signals: mark as CLOSED (the position was already cleaned up).
    Orphan positions: delete (the signal was already closed).
    
    Safe to run on every startup — no-op if DB is consistent.
    """
    with _get_session() as session:
        try:
            # Find all open signal assets+strategies
            open_sig_keys = set(
                (r.asset, r.strategy_name)
                for r in session.query(Signal.asset, Signal.strategy_name)
                .filter(Signal.status == "OPEN")
                .all()
            )

            # Find all position assets+strategies
            pos_keys = set(
                (r.asset, r.strategy_name)
                for r in session.query(
                    OpenPosition.asset,
                    OpenPosition.strategy_name,
                ).all()
            )

            orphan_sigs = open_sig_keys - pos_keys   # signal but no position
            orphan_pos  = pos_keys - open_sig_keys   # position but no signal

            closed_sigs = 0
            deleted_pos = 0

            # Close orphan signals
            for asset, strategy in orphan_sigs:
                sigs = session.query(Signal).filter(
                    Signal.asset          == asset,
                    Signal.strategy_name  == strategy,
                    Signal.status         == "OPEN",
                ).all()
                for sig in sigs:
                    sig.status      = "CLOSED"
                    sig.exit_reason = (
                        "Auto-closed: orphan signal — "
                        "no matching open_positions record found at startup"
                    )
                    logger.info(
                        f"[DB] cleanup_orphan_records: closed orphan signal "
                        f"#{sig.id} | {asset} | {strategy} | {sig.direction}"
                    )
                    closed_sigs += 1

            # Delete orphan positions
            for asset, strategy in orphan_pos:
                pos_rows = session.query(OpenPosition).filter(
                    OpenPosition.asset          == asset,
                    OpenPosition.strategy_name  == strategy,
                ).all()
                for p in pos_rows:
                    logger.info(
                        f"[DB] cleanup_orphan_records: deleted orphan position "
                        f"#{p.id} | {asset} | {strategy} | {p.direction}"
                    )
                    session.delete(p)
                    deleted_pos += 1

            session.commit()

            if closed_sigs or deleted_pos:
                _invalidate_signal_cache()

            result = {
                "orphan_signals_closed":   closed_sigs,
                "orphan_positions_deleted": deleted_pos,
            }
            logger.info(f"[DB] cleanup_orphan_records: {result}")
            return result

        except Exception as e:
            session.rollback()
            logger.error(f"[DB] cleanup_orphan_records failed: {e}")
