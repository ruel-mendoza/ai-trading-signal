PROBLEM STATEMENT
=============================================================

EUR/USD shows an active BUY signal even though:
1. H1 close (1.14668) is below H4 EMA50 (1.15349) — LONG exit triggered
2. D1 conditions have flipped BEARISH — SHORT conditions are met
3. The SHORT signal cannot fire because BUY is still blocking it

Root cause applies to ALL strategies:

STRATEGY          | check_exits() called by scheduler? | Standalone method?
------------------|------------------------------------|-------------------
sp500_momentum    | YES — via run_sp500_intraday_cycle | YES
highest_lowest_fx | YES — explicit call in scheduler   | YES
trend_forex       | YES — explicit call in scheduler   | YES
trend_non_forex   | YES — explicit call in scheduler   | YES
mtf_ema           | NO — MISSING from scheduler job    | NO — MISSING

MultiTimeframeEMAStrategy has NO standalone check_exits() method.
Exit logic only runs inside evaluate() per-asset, and ONLY when
open_position_data is passed in by the caller. If the scheduler
skips an asset or the position is stale, the exit NEVER fires.

Secondary issue: cross-strategy open signals for the same asset
block new entries from other strategies because
has_any_open_signal_for_asset() returns True for the blocking signal
regardless of which strategy owns it, but no strategy's check_exits()
will close another strategy's position.

def check_exits(self) -> list[dict]:
    """
    Standalone exit checker called by the scheduler after the per-asset
    evaluation loop. Ensures exits fire even if evaluate() was skipped.
    Uses the same H1/H4-EMA50 exit logic as _check_h1_ema50_exit().
    """
    closed = []
    positions = get_all_open_positions(strategy_name=STRATEGY_NAME)

    if not positions:
        logger.info(
            "[MTF-EMA-EXIT] ====== check_exits | 0 open positions ======"
        )
        return closed

    logger.info(
        f"[MTF-EMA-EXIT] ====== check_exits | "
        f"{len(positions)} open position(s) ======"
    )

    for pos in positions:
        asset     = pos["asset"]
        pos_id    = pos["id"]
        direction = pos["direction"]

        logger.info(
            f"[MTF-EMA-EXIT] Position #{pos_id} | {asset} | "
            f"{direction} | entry={pos['entry_price']}"
        )

        # Fetch H1 candles for current price
        try:
            h1_candles = self.cache.get_candles(asset, TIMEFRAME_H1, 5)
        except Exception as e:
            logger.error(
                f"[MTF-EMA-EXIT] Position #{pos_id} | {asset} | "
                f"Exception fetching H1 candles: {e}"
            )
            continue

        if not h1_candles:
            logger.warning(
                f"[MTF-EMA-EXIT] Position #{pos_id} | {asset} | "
                f"No H1 candles available — skipping"
            )
            continue

        # Fetch H4 candles for EMA50
        try:
            h4_candles = self.cache.get_candles(asset, TIMEFRAME_H4, 300)
        except Exception as e:
            logger.error(
                f"[MTF-EMA-EXIT] Position #{pos_id} | {asset} | "
                f"Exception fetching H4 candles: {e}"
            )
            continue

        if len(h4_candles) < EMA_50:
            logger.warning(
                f"[MTF-EMA-EXIT] Position #{pos_id} | {asset} | "
                f"Insufficient H4 candles for EMA50: {len(h4_candles)}"
            )
            continue

        h4_closes = [float(c["close"]) for c in h4_candles]
        h4_ema50_series = IndicatorEngine.ema(h4_closes, EMA_50)
        h4_ema50 = (
            h4_ema50_series[-1]
            if h4_ema50_series and h4_ema50_series[-1] is not None
            else None
        )

        if h4_ema50 is None:
            logger.warning(
                f"[MTF-EMA-EXIT] Position #{pos_id} | {asset} | "
                f"H4 EMA50 returned None — skipping"
            )
            continue

        h1_close = float(h1_candles[-1]["close"])

        logger.info(
            f"[MTF-EMA-EXIT] Position #{pos_id} | {asset} | {direction} | "
            f"H1_close={h1_close:.5f} | H4_EMA50={h4_ema50:.5f} | "
            f"LONG_exit(H1<EMA50)={h1_close < h4_ema50} | "
            f"SHORT_exit(H1>EMA50)={h1_close > h4_ema50}"
        )

        exit_triggered = False
        exit_type      = ""

        if direction == "BUY" and h1_close < h4_ema50:
            breach = h4_ema50 - h1_close
            exit_type = (
                f"H1/H4-EMA50 exit (standalone): "
                f"H1 close {h1_close:.5f} < H4 EMA50 {h4_ema50:.5f} "
                f"(breach={breach:.5f})"
            )
            exit_triggered = True
            logger.info(
                f"[MTF-EMA-EXIT] Position #{pos_id} | EXIT LONG | "
                f"H1 {h1_close:.5f} < H4_EMA50 {h4_ema50:.5f}"
            )

        elif direction == "SELL" and h1_close > h4_ema50:
            breach = h1_close - h4_ema50
            exit_type = (
                f"H1/H4-EMA50 exit (standalone): "
                f"H1 close {h1_close:.5f} > H4 EMA50 {h4_ema50:.5f} "
                f"(breach={breach:.5f})"
            )
            exit_triggered = True
            logger.info(
                f"[MTF-EMA-EXIT] Position #{pos_id} | EXIT SHORT | "
                f"H1 {h1_close:.5f} > H4_EMA50 {h4_ema50:.5f}"
            )

        if exit_triggered:
            self._persist_exit(asset, exit_type)
            closed.append({
                **pos,
                "exit_price": h1_close,
                "exit_reason": "h1_ema50_cross",
            })
        else:
            logger.info(
                f"[MTF-EMA-EXIT] Position #{pos_id} | {asset} | "
                f"Holding {direction} — no exit triggered"
            )

    logger.info(
        f"[MTF-EMA-EXIT] ====== check_exits complete | "
        f"{len(closed)} position(s) closed ======"
    )
    return closed

def has_any_open_signal_for_asset(asset: str) -> bool:
    """
    Returns True if ANY open signal exists for this asset across
    all strategies. Used as the primary cross-strategy idempotency
    guard to prevent simultaneous signals on the same asset.
    """
    with _get_session() as session:
        row = (
            session.query(Signal.id, Signal.strategy_name, Signal.direction)
            .filter(
                Signal.asset == asset,
                Signal.status == "OPEN",
            )
            .first()
        )
        if row:
            logger.debug(
                f"[DB] has_any_open_signal_for_asset({asset}): "
                f"blocked by signal #{row[0]} | "
                f"strategy={row[1]} | direction={row[2]}"
            )
            return True
        return False

def close_orphaned_signals_and_positions() -> dict:
    """
    Universal cleanup: find and fix all assets where open signals
    and open positions are mismatched or stale.

    Cases handled:
    1. Signal is OPEN but no matching open_position record exists
       → close the signal (orphaned signal)
    2. open_position record exists but no matching OPEN signal
       → delete the position (orphaned position)
    3. Multiple OPEN signals for the same asset
       → keep highest id, close the rest
    4. Multiple open_position records for the same asset+strategy
       → keep highest id, delete the rest

    This is idempotent — safe to run on every startup.
    """
    from collections import defaultdict

    closed_sigs  = 0
    deleted_pos  = 0
    results      = []

    with _get_session() as session:
        try:
            # ── Case 3: Multiple OPEN signals per asset ──
            all_open_sigs = session.query(Signal).filter(
                Signal.status == "OPEN"
            ).order_by(Signal.asset, Signal.id.desc()).all()

            sigs_by_asset = defaultdict(list)
            for s in all_open_sigs:
                sigs_by_asset[s.asset].append(s)

            for asset, sigs in sigs_by_asset.items():
                if len(sigs) <= 1:
                    continue
                # Keep highest id (most recent), close the rest
                keep = sigs[0]
                for dup in sigs[1:]:
                    dup.status = "CLOSED"
                    dup.exit_reason = (
                        f"Auto-closed: duplicate open signal for {asset} — "
                        f"kept signal #{keep.id} ({keep.strategy_name})"
                    )
                    closed_sigs += 1
                    results.append(
                        f"Closed duplicate signal #{dup.id} "
                        f"{asset} {dup.direction} {dup.strategy_name}"
                    )
                    logger.info(
                        f"[DB] close_orphaned: closed duplicate signal #{dup.id} | "
                        f"asset={asset} | kept #{keep.id}"
                    )

            session.commit()

            # ── Case 4: Multiple open_position per asset+strategy ──
            all_pos = session.query(OpenPosition).order_by(
                OpenPosition.asset,
                OpenPosition.strategy_name,
                OpenPosition.id.desc(),
            ).all()

            pos_by_key = defaultdict(list)
            for p in all_pos:
                pos_by_key[(p.asset, p.strategy_name)].append(p)

            for key, pos_list in pos_by_key.items():
                if len(pos_list) <= 1:
                    continue
                keep_pos = pos_list[0]
                for dup_pos in pos_list[1:]:
                    logger.info(
                        f"[DB] close_orphaned: deleting duplicate position "
                        f"#{dup_pos.id} | asset={key[0]} | "
                        f"strategy={key[1]} | kept #{keep_pos.id}"
                    )
                    session.delete(dup_pos)
                    deleted_pos += 1
                    results.append(
                        f"Deleted duplicate position #{dup_pos.id} "
                        f"{key[0]} {key[1]}"
                    )

            session.commit()

            # ── Re-fetch after dedup for orphan checks ──
            all_open_sigs = session.query(Signal).filter(
                Signal.status == "OPEN"
            ).all()
            all_pos = session.query(OpenPosition).all()

            sig_keys = {
                (s.asset, s.strategy_name)
                for s in all_open_sigs
            }
            pos_keys = {
                (p.asset, p.strategy_name)
                for p in all_pos
            }

            # ── Case 1: OPEN signal with no matching position ──
            for s in all_open_sigs:
                key = (s.asset, s.strategy_name)
                if key not in pos_keys:
                    s.status = "CLOSED"
                    s.exit_reason = (
                        "Auto-closed: orphaned signal — "
                        "no matching open_position record found"
                    )
                    closed_sigs += 1
                    results.append(
                        f"Closed orphaned signal #{s.id} "
                        f"{s.asset} {s.direction} {s.strategy_name}"
                    )
                    logger.info(
                        f"[DB] close_orphaned: closed orphaned signal #{s.id} | "
                        f"asset={s.asset} | strategy={s.strategy_name} | "
                        f"direction={s.direction}"
                    )

            session.commit()

            # ── Case 2: open_position with no matching OPEN signal ──
            # Re-fetch signals after closure
            all_open_sigs_after = session.query(Signal).filter(
                Signal.status == "OPEN"
            ).all()
            sig_keys_after = {
                (s.asset, s.strategy_name)
                for s in all_open_sigs_after
            }

            for p in all_pos:
                key = (p.asset, p.strategy_name)
                if key not in sig_keys_after:
                    logger.info(
                        f"[DB] close_orphaned: deleting orphaned position "
                        f"#{p.id} | asset={p.asset} | "
                        f"strategy={p.strategy_name} | direction={p.direction}"
                    )
                    session.delete(p)
                    deleted_pos += 1
                    results.append(
                        f"Deleted orphaned position #{p.id} "
                        f"{p.asset} {p.strategy_name}"
                    )

            session.commit()

            if closed_sigs or deleted_pos:
                _invalidate_signal_cache()

            logger.info(
                f"[DB] close_orphaned_signals_and_positions complete: "
                f"closed_signals={closed_sigs} deleted_positions={deleted_pos}"
            )

        except Exception as e:
            session.rollback()
            logger.error(
                f"[DB] close_orphaned_signals_and_positions failed: {e}",
                exc_info=True,
            )

    return {
        "closed_signals": closed_sigs,
        "deleted_positions": deleted_pos,
        "details": results,
    }

async function runOrphanCleanup() {
    var btn = document.getElementById('btn-orphan-cleanup');
    var resultEl = document.getElementById('orphan-cleanup-result');
    btn.disabled = true;
    btn.textContent = 'Running...';
    resultEl.innerHTML = '';
    try {
        var res = await fetch(BASE + '/admin/api/admin/cleanup-orphans',
                             { method: 'POST' });
        var data = await res.json();
        if (data.success) {
            var color = (data.closed_signals > 0 || data.deleted_positions > 0)
                ? '#f59e0b' : '#22c55e';
            resultEl.innerHTML =
                '<div style="padding:12px;background:rgba(30,41,59,0.7);' +
                'border:1px solid ' + color + '44;border-radius:8px;">' +
                '<div style="color:' + color + ';font-weight:600;margin-bottom:6px;">' +
                'Cleanup complete</div>' +
                '<div style="color:#e2e8f0;font-size:13px;">' +
                'Closed signals: ' + data.closed_signals + ' | ' +
                'Deleted positions: ' + data.deleted_positions +
                '</div>' +
                (data.details.length > 0
                    ? '<div style="margin-top:8px;font-size:12px;color:#94a3b8;">' +
                      data.details.slice(0, 10).join('<br>') + '</div>'
                    : '') +
                '</div>';
        } else {
            resultEl.innerHTML =
                '<span style="color:#ef4444;">Cleanup failed</span>';
        }
    } catch (e) {
        resultEl.innerHTML =
            '<span style="color:#ef4444;">Error: ' + e.message + '</span>';
    }
    btn.disabled = false;
    btn.textContent = 'Run Signal Integrity Cleanup';
}
