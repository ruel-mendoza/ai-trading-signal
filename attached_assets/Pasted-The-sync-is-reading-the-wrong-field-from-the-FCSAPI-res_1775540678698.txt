Replace the current logic with this approach:

Do NOT try to read remaining credits from the FCSAPI response info.credit_count field — that field only contains how many credits the current request consumed (always 1-2), not the account balance.
Instead, calculate the correct remaining credits from the local api_usage_log table which already accurately tracks real usage:

pythondef sync_quota_from_api() -> dict:
    from trading_engine.database import get_setting, set_setting, SessionFactory
    from sqlalchemy import text
    from datetime import datetime, timezone
    
    # Get credit limit
    limit_str = get_setting("fcsapi_credit_limit")
    credit_limit = int(limit_str) if limit_str else 500_000
    
    # Calculate real usage from api_usage_log for current billing month
    now = datetime.now(timezone.utc)
    month_start = now.replace(day=1, hour=0, minute=0, second=0, microsecond=0).isoformat()
    
    with SessionFactory() as session:
        monthly_used = session.execute(
            text("SELECT COALESCE(SUM(credits_used), 0) FROM api_usage_log WHERE timestamp >= :start"),
            {"start": month_start}
        ).scalar() or 0
    
    real_remaining = max(0, credit_limit - monthly_used)
    usage_pct = round((monthly_used / credit_limit) * 100, 2) if credit_limit > 0 else 0
    
    # Update app_settings to match
    set_setting("fcsapi_remaining_credits", str(real_remaining))
    set_setting("fcsapi_credit_limit", str(credit_limit))
    
    # Clear kill switch if usage is below critical threshold (95%)
    kill_switch_cleared = False
    if usage_pct < 95.0:
        set_setting("credit_kill_switch", "false")
        set_setting("quota_watchdog_disabled", "false")
        set_setting("credit_kill_switch_month", "")
        kill_switch_cleared = True
    
    synced_at = now.strftime("%Y-%m-%dT%H:%M:%SZ")
    set_setting("fcsapi_quota_last_updated", now.strftime("%Y-%m-%d %H:%M:%S UTC"))
    
    return {
        "success": True,
        "real_remaining": real_remaining,
        "real_limit": credit_limit,
        "monthly_used": monthly_used,
        "usage_pct": usage_pct,
        "kill_switch_cleared": kill_switch_cleared,
        "synced_at": synced_at,
    }
