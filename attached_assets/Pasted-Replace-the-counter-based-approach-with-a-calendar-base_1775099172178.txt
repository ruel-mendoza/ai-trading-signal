pythondef _count_trading_days(entry_date_str: str) -> int:
    """Count actual trading days elapsed since entry."""
    import pytz
    from datetime import datetime, date
    import numpy as np

    et = pytz.timezone("America/New_York")
    try:
        entry_dt = datetime.strptime(str(entry_date_str)[:19], "%Y-%m-%dT%H:%M:%S")
        entry_date = entry_dt.date()
    except (ValueError, TypeError):
        return 0

    today = datetime.now(pytz.utc).astimezone(et).date()

    # Count business days between entry and today
    # numpy busday_count excludes weekends automatically
    trading_days = int(np.busday_count(entry_date, today))
    return max(0, trading_days)
