PROBLEM SUMMARY
=============================================================

The active signals panel shows duplicate entries for the same asset
where one signal is from a strategy (strategy_name="mtf_ema" etc.)
and another is from an old AI/manual source (different strategy_name
or no strategy_name). Both show as OPEN simultaneously.

Root cause: deduplication keys on (asset + strategy_name), meaning
signals from different strategies for the same asset are both shown.

Correct rule: only ONE open signal per asset across ALL strategies.
If any signal is OPEN for an asset, no other signal for that asset
should appear as active regardless of which strategy created it.

def close_stale_manual_signals():
    """
    One-time cleanup: close any OPEN signals that were created manually
    or by an external AI service before the strategy engine was running.
    Identified by: no matching open position record in open_positions table,
    AND signal is older than 7 days, AND not created by a known strategy.

    Known strategy names: mtf_ema, trend_forex, trend_non_forex,
    sp500_momentum, highest_lowest_fx, trend_following
    """
    KNOWN_STRATEGIES = {
        "mtf_ema",
        "trend_forex",
        "trend_non_forex",
        "sp500_momentum",
        "highest_lowest_fx",
        "trend_following",
    }
    cutoff = (datetime.utcnow() - timedelta(days=7)).isoformat()

    with _get_session() as session:
        try:
            # Find OPEN signals not from known strategies that are older than 7 days
            stale = session.query(Signal).filter(
                Signal.status == "OPEN",
                Signal.created_at <= cutoff,
                Signal.strategy_name.notin_(KNOWN_STRATEGIES),
            ).all()

            if not stale:
                logger.info("[DB] close_stale_manual_signals: no stale signals found")
                return

            for sig in stale:
                sig.status = "CLOSED"
                sig.exit_reason = "Auto-closed: stale manual/AI signal predates strategy engine"
                logger.info(
                    f"[DB] close_stale_manual_signals: closing signal #{sig.id} | "
                    f"asset={sig.asset} | direction={sig.direction} | "
                    f"strategy={sig.strategy_name} | created={sig.created_at}"
                )

            session.commit()
            logger.info(
                f"[DB] close_stale_manual_signals: closed {len(stale)} stale signal(s)"
            )
            _invalidate_signal_cache()

        except Exception as e:
            session.rollback()
            logger.error(f"[DB] close_stale_manual_signals failed: {e}")

def has_any_open_signal_for_asset(asset: str) -> bool:
    """
    Returns True if ANY open signal exists for this asset,
    regardless of strategy_name or direction.
    Use this as the primary idempotency guard to prevent cross-strategy
    duplicates (e.g. MTF EMA trying to BUY while a manual SELL is open).
    """
    with _get_session() as session:
        row = (
            session.query(Signal.id)
            .filter(
                Signal.asset == asset,
                Signal.status == "OPEN",
            )
            .first()
        )
        return row is not None
